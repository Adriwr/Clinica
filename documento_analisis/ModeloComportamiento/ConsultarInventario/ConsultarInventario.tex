\begin{UseCase}{CU23}{Consultar Inventario}{El cajero consultará la lista completa de productos registrados en el sistema. El sistema mostrará la lista de productos.}
	\UCitem{Versión}{0.2}
    \UCitem{Actor}{Cajero}
    \UCitem{Propósito}{Mostrar los productos registrados. Poder modificar el inventario}
    \UCitem{Entradas}{Ninguna.}
    \UCitem{Origen}{}
    \UCitem{Salidas}{
        \begin{itemize}
	        \item Lista de productos: Arreglo de Estructuras de datos que contienen:
	        \begin{itemize}
	        	\item ID del producto: cadena de caracteres alfanuméricos.
	        	\item Nombre: Cadena de caracteres.
	        	\item Precio: Número flotante positivo con dos decimales.
	        	\item Existencias: Número entero positivo.
	        \end{itemize}
		\end{itemize}
	}
    \UCitem{Destino}{
		\begin{itemize}
			\item Nombre, Precio, Existencias: Pantalla IU23 Consultar Inventario.
			\item ID del producto: oculto.
		\end{itemize}
	}
	
   	\UCitem{Precondiciones}{
   	    \begin{itemize}
			\item Existen productos registrados en el sistema.
   	    \end{itemize}
   	}
    \UCitem{Postcondiciones}{Ninguna}
    \UCitem{Observaciones}{Ninguna}
    \UCitem{Errores}{
    	\begin{itemize}
    	\item ERR1 - No hay productos registrados en el repositorio de datos:
	    	\begin{itemize}
	    		\item Muestra el mensaje MSG23a ``No existen productos registrados''.
	    		\item Fin del caso de uso.
	    	\end{itemize}
		\end{itemize}
	}
	\UCitem{Tipo de ejecución}{Secundaria, viene de CU1 Iniciar sesión.}
	\UCitem{Prioridad}{Alta}
	\UCitem{Volatilidad}{Baja}
	\UCitem{Madurez}{Alta}
	\UCitem{Estado}{Terminado}
	\UCitem{Autor}{Demis Gómez Moncada}
	\UCitem{Revisor}{Adrián Eduardo Galindo García}
	
\end{UseCase}

\begin{UCtrayectoria}{}
		\UCpaso[\UCactor] selecciona consultar el inventario dando click en el botón ``Consultar Inventario''.
		\UCpaso solicita los datos de los productos registrados al repositorio de datos. [ERR1].
		\UCpaso muestra la pantalla IU23 Consultar inventario.
\end{UCtrayectoria}
